SCRIPT thugplus_levelmenu_remove_hud 
	IF ScreenElementExists id = thugplus_levelselect_hud 
		DestroyScreenElement id = thugplus_levelselect_hud 
	ENDIF 
ENDSCRIPT

SCRIPT thugplus_levelmenu_create_hud 
	IF ScreenElementExists id = thugplus_levelselect_hud 
		DestroyScreenElement id = thugplus_levelselect_hud 
	ENDIF 
	CreateScreenElement { 
		type = ContainerElement 
		id = thugplus_levelselect_hud 
		parent = current_menu_anchor 
		dims = PAIR(150, 50) 
		pos = PAIR(490, 100) 
		z_priority = 18 
	} 
	CreateScreenElement { 
		type = SpriteElement 
		parent = thugplus_levelselect_hud 
		texture = left_arrow 
		rgba = [ 90 90 100 180 ] 
		pos = PAIR(-20, 0) 
		scale = PAIR(1.25, 0.8) 
		alpha = 0.60000002384 
		z_priority = 8 
	} 
	CreateScreenElement { 
		type = SpriteElement 
		parent = thugplus_levelselect_hud 
		texture = right_arrow 
		rgba = [ 90 90 100 180 ] 
		pos = PAIR(150.00000000000, 0.00000000000) 
		scale = PAIR(1.25000000000, 0.80000001192) 
		alpha = 0.60000002384 
		z_priority = 8 
	} 
	CreateScreenElement { 
		type = TextElement 
		parent = thugplus_levelselect_hud 
		font = small 
		text = <level_list_name> 
		scale = 1.10000002384 
		pos = PAIR(60.00000000000, 0.00000000000) 
		just = [ center center ] 
		rgba = [ 90 90 100 180 ] 
		not_focusable 
	} 
ENDSCRIPT

SCRIPT thugplus_levelmenu_left 
	printf "thugplus_levelmenu_left" 
	change THUGPLUS_LEVEL_INDEX = ( THUGPLUS_LEVEL_INDEX - 1 ) 
	IF ( THUGPLUS_LEVEL_INDEX < 0 ) 
		change THUGPLUS_LEVEL_INDEX = THUGPLUS_LEVEL_PAGELIMIT 
	ENDIF 
	thugplus_levelmenu_create 
ENDSCRIPT

SCRIPT thugplus_levelmenu_right 
	printf "thugplus_levelmenu_right" 
	change THUGPLUS_LEVEL_INDEX = ( THUGPLUS_LEVEL_INDEX + 1 ) 
	IF ( THUGPLUS_LEVEL_INDEX > THUGPLUS_LEVEL_PAGELIMIT ) 
		change THUGPLUS_LEVEL_INDEX = 0 
	ENDIF 
	thugplus_levelmenu_create 
ENDSCRIPT

SCRIPT thugplus_levelmenu_create 
	level_list = THUGPLUS_LEVELS_UG1 
	list_name = "THUG1" 
	SWITCH THUGPLUS_LEVEL_INDEX 
		CASE 0 
			<level_list> = THUGPLUS_LEVELS_UG1 
			<list_name> = "THUG1" 
		CASE 1 
			<level_list> = THUGPLUS_LEVELS_TH1 
			<list_name> = "THPS1" 
		CASE 2 
			<level_list> = THUGPLUS_LEVELS_TH2 
			<list_name> = "THPS2" 
		CASE 3 
			<level_list> = THUGPLUS_LEVELS_TH3 
			<list_name> = "THPS3" 
		CASE 4 
			<level_list> = THUGPLUS_LEVELS_TH4 
			<list_name> = "THPS4" 
		CASE 5 
			<level_list> = THUGPLUS_LEVELS_UG2 
			<list_name> = "THUG2" 
		CASE 6 
			<level_list> = THUGPLUS_LEVELS_AW 
			<list_name> = "THAW" 
		CASE 7 
			<level_list> = THUGPLUS_LEVELS_P8 
			<list_name> = "THP8" 
	ENDSWITCH 
	change last_level_list = <level_list> 
	change last_level_list_name = <list_name> 
	level_select_menu_exit no_unload 
	launch_level_select_menu use_level_list = <level_list> level_list_name = <list_name> dont_wait no_unload 
ENDSCRIPT
